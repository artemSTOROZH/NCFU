CREATE TABLE DismissalArticle(
ArticleID INT PRIMARY KEY,
Name TEXT,
DismissalReason TEXT,
Number char(4),
ParagraphNumber char(4)
)

CREATE TABLE Employee(
EmpID INT PRIMARY KEY,
LastName nchar(25),
FirstName nchar(20),
MiddleName nchar(20),
JobTitle nchar(25),
DepartmentName nchar(30),
EnterDate date
)

CREATE TABLE DismissalDocument(
DocumentID INT PRIMARY KEY,
Number nchar(4),
RegDate date,
DismissalDate date,
ArticleID INT,
EmpID INT,
Compensation DECIMAL(10, 0)
)
